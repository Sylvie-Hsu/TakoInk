package main
import(
  "fmt"
  "strconv"
  "github.com/hyperledger/fabric/core/chaincode/shim"
  pb "github.com/hyperledger/fabric/protos/peer"
)

type SimpleChaincode struct {}


func (t *SimpleChaincode) Init(stub shim.ChaincodeStubInterface) pb.Response {
  fmt.Println("ex02 Init")
  var A string
  var Aval int
  var err_A error
  A = "A"
  Aval,err_A = strconv.Atoi("500")
  if err_A!= nil {
  	return shim.Error("Expecting integer value for asset holding")
  }
  err_A = stub.PutState(A, []byte(strconv.Itoa(500)))
  if err_A != nil {
    return shim.Error(err_A.Error())
  }
  var B string
  var Bval int
  var err_B error
  B = "B"
  Bval,err_B = strconv.Atoi("1000")
  if err_B!= nil {
  	return shim.Error("Expecting integer value for asset holding")
  }
  err_B = stub.PutState(B, []byte(strconv.Itoa(1000)))
  if err_B != nil {
    return shim.Error(err_B.Error())
  }
    return shim.Success(nil)
}
func (t *SimpleChaincode) Invoke(stub *shim.ChaincodeStub, function string, args []string) ([]byte, error) {
    fmt.Println("ex02 Invoke")
    if function == "invoke" {
      fmt.Printf("Function is invoke")
      return t.invoke(stub, args)
    } else if function == "init" {
      fmt.Printf("Function is init")
      return t.Init(stub, function, args)
    } else if function == "delete" {
      fmt.Printf("Function is delete")
      return t.delete(stub, args)
    } else if function == "query" {
      fmt.Printf("Function is query")
      return t.query(stub, args)
    }
    return shim.Error("Invalid invoke function name. Expecting \"invoke\" \"delete\" \"query\"")
}
  func (t *SimpleChaincode) invoke(stub shim.ChaincodeStub, args []string) ([]byte, error) {
      fmt.Printf("Running invoke")
      var USER_A, USER_B string
      var Aval, Bval int
      var MONEY int
      var err error
      USER_A = "A"
      USER_B = "B"
      USER_A_val_bytes, err := stub.GetState(USER_A)
      if err != nil {
        return shim.Error("Failed to get state")
      }
      if USER_A_val_bytes == nil {
        return shim.Error("Entity not found")
      }
      Aval, _ = strconv.Atoi(string(USER_A_val_bytes))
      USER_B_val_bytes, err := stub.GetState(USER_B)
      if err != nil {
        return shim.Error("Failed to get state")
      }
      if USER_B_val_bytes == nil {
        return shim.Error("Entity not found")
      }
      Bval, _ = strconv.Atoi(string(USER_B_val_bytes))
      MONEY, err = strconv.Atoi("50")
      Aval = Aval -50
      Bval = Bval +50
      fmt.Printf("Aval = %d, Bval = %d", Aval, Bval)
      err = stub.PutState(USER_A, []byte(strconv.Itoa(Aval)))
      if err != nil {
        return shim.Error(err.Error())
      }
      err = stub.PutState(USER_B, []byte(strconv.Itoa(Bval)))
      if err != nil {
        return shim.Error(err.Error())
      }
      return shim.Success(nil)
  }
  func (t *SimpleChaincode) Query(stub shim.ChaincodeStub, args []string) pb.Response {
      fmt.Printf("Running query")
      var user_name string
      var err error
      user_name = "A"
      user_name_val, err := stub.GetState(user_name)
      if err != nil {
        jsonResp := "{\"Error\":\"Failed to get state for " + user_name + "\"}"
        return shim.Error(jsonResp)
      }
      if user_name_val == nil {
        jsonResp := "{\"Error\":\"Nil amount for " + user_name + "\"}"
        return shim.Error(jsonResp)
      }
      jsonResp := "{\"Name\":\"" + user_name + "\",\"Amount\":\"" + string(user_name_val) + "\"}"
      fmt.Printf("Query Response:%s\n", jsonResp)
      return shim.Success(user_name_val)
  }
  func (t *SimpleChaincode) delete(stub shim.ChaincodeStub, args []string) ([]byte, error) {
      fmt.Printf("Running delete")
      user_name := "B"
      err := stub.DelState(user_name)
      if err != nil {
         return shim.Error("Failed to delete state")
      }
      return shim.Success(nil)
  }
func main() {
  err := shim.Start(new(SimpleChaincode))
  if err != nil {
    fmt.Printf("Error starting Simple chaincode: %s", err)
  }
}
